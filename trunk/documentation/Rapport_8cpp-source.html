00161 void Rapport::ecritLignups(FileHandle_t & fh, Match & match)
00162 {
00163         Team * team1 = match.getTeam1();
00164         Team * team2 = match.getTeam2();
00165 
00166         const string * tagT;
00167         const string * tagCT;
00168         if (match.getMancheCourante() == 1)
00169         {
00170                 tagT = &team1->getTag();
00171                 tagCT = &team2->getTag();
00172         }
00173         else
00174         {
00175                 tagT = &team2->getTag();
00176                 tagCT = &team1->getTag();
00177         }
00178 
00179         vector<int> lignupT = vector<int>();
00180         vector<int> lignupCT = vector<int>();
00181         vector<int> lignupSpec = vector<int>();
00182 
00183         for (int i=1;i<=API::maxplayers;i++)
00184         {
00185                 edict_t * pEntity = API::engine->PEntityOfEntIndex(i);
00186                 try
00187                 {
00188                         IPlayerInfo * player = API::getIPlayerInfoFromEntity(pEntity);
00189                         switch(player->GetTeamIndex())
00190                         {
00191                         case TEAM_T:
00192                                 lignupT.push_back(i);
00193                                 break;
00194                         case TEAM_CT:
00195                                 lignupCT.push_back(i);
00196                                 break;
00197                         default:
00198                                 lignupSpec.push_back(i);
00199                         }
00200                 }
00201                 catch(const out_of_range & e){}
00202         }
00203 
00204         Rapport::ecritLignupT(fh,match,lignupT,tagT);
00205         Rapport::ecritLignupCT(fh,match,lignupCT,tagCT);
00206         Rapport::ecritLignupSpec(fh,match,lignupSpec);
