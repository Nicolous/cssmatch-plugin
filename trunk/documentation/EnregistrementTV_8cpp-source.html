00038 EnregistrementTV::EnregistrementTV(const string & fichierSansPathSansExtension)
00039 {
00040         // Si sourceTV n'est pas en mesure d'enregistrer (absente ou tv_enable 0) on jette une exception de type std::out_of_range
00041         if (!ConVars::tv_enable)
00042                 throw out_of_range("Impossible de lancer l'enregistrement sourceTV car tv_enable est introuvable");
00043         if (!ConVars::tv_enable->GetBool())
00044                 throw out_of_range("Impossible de lancer l'enregistrement sourceTV car tv_enable est sur 0");
00045         if (!API::hltvPresente())
00046                 throw out_of_range("Impossible de lancer l'enregistrement sourceTV car celle-ci n'est pas sur le serveur");
00047 
00048         // On s'alloue une copie modifiable de la variable
00049         string fichier = fichierSansPathSansExtension;
00050 
00051         // On supprime les caractères génant pour Windows ou pour la console
00052         API::valideNomDeFichier(fichier);
00053 
00054         // On construit le path relatif du fichier
00055         string path = ConVars::cssmatch_sourcetv_path.GetString();
00056         if (path != "")
00057                 fichier = path + "/" + fichier + ".dem";
00058 
00059         const char * fichierC = fichier.c_str();
00060 
00061         // Pour faciliter le passage par les méthodes de VALVE, nous repassons temporairement en char*...
00062         char buffer[512];
00063         V_snprintf(buffer,sizeof(buffer),"%s",fichierC);
00064 
00065         // Tant que le nom du fichier formé existe déjà, nous lui ajoutons un indice
00066         int nbrFichier = 1;
00067         while (API::filesystem->FileExists(buffer, "MOD"))
00068         {
00069                 V_snprintf(buffer,sizeof(buffer),"[%i]%s",nbrFichier,fichierC);
00070                 nbrFichier++;
00071         }
00072 
00073         // On récupère le nom final du fichier dans le champ nom
00074         nom = buffer;
00075 }

00077 void EnregistrementTV::enregistre() const
00078 {
00079         // Nous préparons la commande
00080         string cmd = "tv_record " + nom + "\n";
00081 
00082         // Nous lançons l'enregistrement en stoppant tout autre enregistrement en cours
00083         API::serveurExecute("tv_stoprecord\n");
00084         API::serveurExecute("tv_autorecord 0\n");
00085         API::serveurExecute(cmd);
