00028 void Messages::sayMsg(const string & message, int indexJoueur)
00029 {
00030         // Chaque message du TCHAT sera préfixé par [CSSMatch] :
00031         string format = "\004[CSSMatch] :\001 " + message + "\n";
00032         const char * msg = format.c_str();
00033 
00034         // Préparation du message
00035         MRecipientFilter filter;
00036         filter.AddAllPlayers();
00037         bf_write * pBitBuf = API::engine->UserMessageBegin( &filter, MESSAGE_SAYTEXT );
00038         // Ici l'index d'un joueur permet de colorer une partie du message précédée par un \003 de la couleur de la team du joueur
00039         // CT <=> bleu
00040         // T <=> rouge
00041         // SPEC/UN <=> blanc
00042         pBitBuf->WriteByte(indexJoueur);
00043         pBitBuf->WriteString(msg);
00044         // API::engine->ServerCommand(("say " + message + "\n").str());
00045         API::engine->LogPrint(msg);
00046         pBitBuf->WriteByte(1);
00047         API::engine->MessageEnd();
00048 }

00095 void Messages::tellAMX(int indexJoueur, const string & message, int vie)
00096 {
00097         const char * msg = message.c_str();
00098         // Nous devons éviter que le joueur choisisse par accident une option dans un menu du plugin caché par une fenêtre AMX
00099         GestionJoueurs::getPlayerList()->getJoueur(indexJoueur)->getMenu()->setCodeMenu(CODE_MENU_INVALIDE);
00100         //API::cexec(API::engine->PEntityOfEntIndex(indexJoueur),"slot10\n");
00101         MRecipientFilter filter;
00102         filter.AddRecipient(indexJoueur);
00103         bf_write *pBuffer = API::engine->UserMessageBegin( &filter, MESSAGE_SHOWMENU );
00104         // Toutes les touches permetteront d'effacer le menu
00105         pBuffer->WriteShort(OPTION_ALL);
00106         pBuffer->WriteChar(vie);
00107         pBuffer->WriteByte(false);
