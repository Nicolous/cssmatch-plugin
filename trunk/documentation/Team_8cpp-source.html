00112         // Il n'y a personne dans le camp désigné par le codeTeam

00114         {

00170         // Variable buffer contenant l'entité actuellement pointée
00171         edict_t * player = NULL;

00173         // Variables contenant les 2 joueurs trouvés
00174         // La recherche d'un bout de pseudo commun débouchera sur le tag supposé de la team

00179         IPlayerInfo * infoJ1 = NULL;
00180         IPlayerInfo * infoJ2 = NULL;
00181 
00182         // On cherche à récupérer 2 des joueurs de le camp, ainsi que le pointeur sur IPlayerInfo qui leur est associé
00183         for (int i=1;i<=API::maxplayers;i++)
00184         {
00185                 // On récupère l'entité associé à cet index i
00186                 player = API::engine->PEntityOfEntIndex(i);
00187                 // Si l'entité est invalide, on repasse player à NULL et on ne fait rien d'autre
00188                 IPlayerInfo * playerinfo;
00189                 try
00190                 {
00191                         playerinfo = API::getIPlayerInfoFromEntity(player);
00192                         if (playerinfo->GetTeamIndex()==codeTeam)
00193                         {
00194                                 // On a trouvé notre j1 si ce n'était pas encore le cas
00195                                 if (!j1)
00196                                 {
00197                                         j1 = player;
00198                                         infoJ1 = playerinfo;
00199                                 }
00200                                 // Sinon on a trouvé notre j2
00201                                 else
00202                                 {
00203                                         j2 = player;
00204                                         infoJ2 = playerinfo;
00205                                         // Dans ce cas on a nos 2 joueurs, on peut stopper la boucle proprement
00206                                         i = API::maxplayers+1;
00207                                 }
00208                         }
00209                 }
00210                 catch(const out_of_range & e)
00211                 {
00212                         player = NULL;
00213                         playerinfo = NULL;

00215         }
00216         
00217         // On récupère le pseudo des 2 joueurs trouvés
00218         const string pseudoJ1 = infoJ1->GetName();

00221         // On récupère la taille des 2 pseudos
