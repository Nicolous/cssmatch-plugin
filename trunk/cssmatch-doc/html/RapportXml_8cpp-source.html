00037 
00038         size_t debutRecherche = 0;
00039 
00040         while((debutRecherche = xml.find("&",debutRecherche)) != string::npos)
00041         {
00042                 xml.replace(debutRecherche,1,"&amp;",0,5);

00046         while((debutRecherche = xml.find("<",debutRecherche)) != string::npos)

00048                 xml.replace(debutRecherche,1,"&lt;",0,4);
