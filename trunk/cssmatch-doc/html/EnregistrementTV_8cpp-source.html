00041 EnregistrementTV::EnregistrementTV(const string & fichierSansPathSansExtension) throw(CSSMatchTVException)
00042 {
00043         // Si sourceTV n'est pas en mesure d'enregistrer (absente ou tv_enable 0)
00044         //      on jette une exception de type CSSMatchTVException
00045         if (! ConVars::tv_enable)
00046                 throw CSSMatchTVException("Impossible de lancer l'enregistrement sourceTV car tv_enable est introuvable");
00047         if (! ConVars::tv_enable->GetBool())
00048                 throw CSSMatchTVException("Impossible de lancer l'enregistrement sourceTV car tv_enable est sur 0");
00049         if (! Api::hltvPresente())
00050                 throw CSSMatchTVException("Impossible de lancer l'enregistrement sourceTV car celle-ci n'est pas sur le serveur");
00051 
00052         // On s'alloue une copie modifiable de la variable
00053         string fichier(fichierSansPathSansExtension);
00054 
00055         // L'enregistrement n'est pas immédiatement lancé
00056         enregistrant = false;
00057 
00058         // On supprime les caractères génant pour Windows ou pour la console
00059         Api::valideNomDeFichier(fichier);
00060 
00061         // On construit le path relatif du fichier
00062         string path(ConVars::cssmatch_sourcetv_path.GetString());
00063         if (path != "")
00064                 fichier = path + "/" + fichier + ".dem";
00065 
00066         const char * fichierC = fichier.c_str();
00067 
00068         nom = fichierC;
00069 
00070         // Tant que le nom du fichier formé existe déjà, nous lui ajoutons un indice
00071         int nbrFichier = 1;
00072         while (Api::filesystem->FileExists(nom.c_str(), "MOD"))
00073         {
00074                 ostringstream renomme;
00075                 renomme << "[" << nbrFichier << "]" << fichierC;
00076                 nom = renomme.str();
00077                 nbrFichier++;
00078         }
00079 }

00081 void EnregistrementTV::enregistre() throw (CSSMatchTVException)
00082 {
00083         if (nom.empty())
00084                 throw CSSMatchTVException("Impossible de lancer un engistrement SourceTV sans nom");
00085 
00086         string cmd("tv_record " + nom + "\n");
00087 
00088         // Nous lançons l'enregistrement en stoppant tout autre enregistrement en cours
00089         Api::serveurExecute("tv_stoprecord\n");
00090         Api::serveurExecute("tv_autorecord 0\n");
00091         Api::serveurExecute(cmd);
00092 
00093         enregistrant = true;
00094 }
