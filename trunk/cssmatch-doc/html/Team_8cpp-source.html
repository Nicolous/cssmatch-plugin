00129 {
00130         // Variable contenant le nombre de joueurs correspondant au codeTeam
00131         nbrMembres = Api::getPlayerCount(codeTeam);
00132         
00133         // Il n'y a personne dans le camp désigné par le codeTeam
00134         if (nbrMembres == 0)
00135                 aucunTag();
00136 
00137         // Il y a 1 joueur dans le camp
00138         else if (nbrMembres == 1)
00139         {
00140                 // On le recherche et son pseudo fera office de tag

00146                 // Nous sommes maintenant dans le cas où il y a au moins 1 joueur dans le camp

00186 void Team::determineTag(CodeTeam codeTeam)
00187 {
00188         // Variable buffer contenant l'entité actuellement pointée
00189         edict_t * player = NULL;
00190 
00191         // Variables contenant les 2 joueurs trouvés
00192         // La recherche d'un bout de pseudo commun débouchera sur le tag supposé de la team
00193         edict_t * j1 = NULL;
00194         edict_t * j2 = NULL;
00195 
00196         // Variable pointant sur l'interface fournissant diverses infos sur les joueurs j1 et j2 (ex : leur pseudo 0_0)
00197         IPlayerInfo * infoJ1 = NULL;
00198         IPlayerInfo * infoJ2 = NULL;
00199 
00200         // On cherche à récupérer 2 des joueurs de le camp, ainsi que le pointeur sur IPlayerInfo qui leur est associé
00201         for (int i=1;i<=Api::maxplayers;i++)
00202         {
00203                 // On récupère l'entité associé à cet index i
00204                 player = Api::engine->PEntityOfEntIndex(i);
00205                 // Si l'entité est invalide, on repasse player à NULL et on ne fait rien d'autre
00206                 IPlayerInfo * playerinfo = NULL;
00207                 try
00208                 {
00209                         playerinfo = Api::getIPlayerInfoFromEntity(player);
00210                         if (playerinfo->GetTeamIndex() == codeTeam)
00211                         {
00212                                 // On a trouvé notre j1 si ce n'était pas encore le cas
00213                                 if (j1 == NULL)
00214                                 {
00215                                         j1 = player;
00216                                         infoJ1 = playerinfo;
00217                                 }
00218                                 // Sinon on a trouvé notre j2
00219                                 else
00220                                 {
00221                                         j2 = player;
00222                                         infoJ2 = playerinfo;
00223                                         // Dans ce cas on a nos 2 joueurs, on peut stopper la boucle proprement
00224                                         i = Api::maxplayers + 1;
00225                                 }
00226                         }
00227                 }
00228                 catch(const CSSMatchApiException & e)
00229                 {
00230                         player = NULL;
00231                         playerinfo = NULL;
00232                 }
00233         }
00234         
00235         // On récupère le pseudo des 2 joueurs trouvés
00236         const string pseudoJ1(infoJ1->GetName());
00237         const string pseudoJ2(infoJ2->GetName());
00238 
00239         // On récupère la taille des 2 pseudos
00240         int tailleJ1 = pseudoJ1.size();
00241         int tailleJ2 = pseudoJ2.size();
00242 
00243         // Dès que l'on trouve un caractère commun dans les pseudos
00244         for (int i=0;i<tailleJ1;i++)
00245         {
00246                 // Pour celà on compare chaque lettre du pseudo du joueur avec chaque lettre du pseudo du joueur 2
00247         int j = 0; // On souhaite que j est une durée de vie supérieure à celle de la boucle, donc on fera un while
00248                 // Tant que les caractères comparés ne sont pas égaux on incrémente j
00249 
00250                 while(j<tailleJ2 && pseudoJ1[i]!=pseudoJ2[j])
00251                         j++;

00253         if (j<tailleJ2 && confirmeTag(pseudoJ1,i,pseudoJ2,j))
00254             i = tailleJ1;
00255     }
