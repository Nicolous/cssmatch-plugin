00027 void PlayerProp::getClasseOffet() throw(CSSMatchPlayerPropException)
00028 {
00029         // Récupération d'un pointeur sur la liste des classes
00030         ServerClass * classes = Api::serverDll->GetAllServerClasses();
00031 
00032         // Recherche de la classe
00033         while (classes != NULL)
00034         {
00035                 if (classe == classes->GetName())
00036                 {
00037                         // Recherche de l'offset
00038                         int nbrProps = classes->m_pTable->GetNumProps();
00039                         for(int i=0;i<nbrProps;i++)
00040                         {
00041                                 SendProp * sProp = classes->m_pTable->GetProp(i);
00042                                 if (prop == sProp->GetName())
00043                                 {
00044                                         // L'offset doit être positif
00045                                         offset = sProp->GetOffset();
00046                                         if (offset < 0)
00047                                                 offset *= -1;
00048 
00049                                         i = nbrProps; // arrêt de la boucle
00050                                 }
00051                         }
00052                 }
00053 
00054                 classes = classes->m_pNext;
00055         }
00056 
00057         if (offset == 0)
00058                 throw CSSMatchPlayerPropException(string("Impossible de trouver l'offset de la prop ") + classe + "." + prop + " !");
00059 }
