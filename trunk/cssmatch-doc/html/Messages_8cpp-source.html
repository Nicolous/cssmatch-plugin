00028 void Messages::sayMsg(const string & message, int indexJoueur)
00029 {
00030         // Chaque message du TCHAT sera préfixé par [CSSMatch] :
00031         string format("\004[" CSSMATCH_NOM "] :\001 " + message + "\n");
00032         const char * msg = format.c_str();
00033 
00034         MRecipientFilter filter;
00035         filter.addAllPlayers();
00036         bf_write * pBitBuf = Api::engine->UserMessageBegin( &filter, MESSAGE_SAYTEXT );
00037 
00038         if (pBitBuf != NULL)
00039         {
00040                 // Ici l'index d'un joueur permet de colorer une partie du message précédée par un \003
00041                 //              de la couleur de la team du joueur
00042                 // CT <=> bleu
00043                 // T <=> rouge
00044                 // SPEC/UN <=> blanc
00045                 pBitBuf->WriteByte(indexJoueur);
00046                 pBitBuf->WriteString(msg);
00047                 Api::engine->LogPrint(msg);
00048                 pBitBuf->WriteByte(1);
00049                 Api::engine->MessageEnd();
00050         }
00051         else
00052                 Api::debug("Type d'UserMessage invalide");
00053 }

00105 void Messages::tellAMX(int indexJoueur, const string & message, int vie)
00106 {
00107         // Si le message fait plus de 1024 caractères le message est invalide
00108         if (message.size() <= MENU_TAILLE_LIMITE)
00109         {
00110                 const char * msg = message.c_str();
00111                 // Nous devons éviter que le joueur choisisse par accident une option dans un menu du plugin caché par une fenêtre AMX
