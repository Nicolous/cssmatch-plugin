00033 void afficheDecompte(const string & parametre)
00034 {
00035         // Calcul du temps en minutes et secondes
00036         Decompte * decompte = Decompte::getDecompte();
00037         int tempsRestant = decompte->getTempsRestant();
00038         int secondes = tempsRestant;
00039         int minutes = (int)(secondes/60);
00040         secondes -= minutes*60;
00041 
00042         // On affiche le nombre de minutes et de secondes restantes
00043         ostringstream msg;
00044         if (secondes >= 10)
00045         {
00046                 if (minutes >= 10)
00047                         msg << minutes << " : " << secondes;
00048                 else
00049                         msg << "0" <<minutes << " : " << secondes;
00050         }
00051         else
00052         {
00053                 if (minutes >= 10)
00054                         msg << minutes << " : " << "0" << secondes;
00055                 else
00056                         msg << "0" << minutes << " : " << "0" << secondes;
00057         }
00058         Messages::hintMsg(msg.str());
00059 
00060         // S'il reste du temps à décompter, on planifie le prochain timer
00061         if (decompte->decTempsRestant()>=0)
00062         {
00063                 decompte->getMoteur()->addTimer(Timer(1.0f,afficheDecompte));
00064         }
00065         else
00066                 // Sinon on stoppe le décompte
00067                 decompte->finDecompte();
00068 }

00122 void Decompte::finDecompte()
00123 {
00124         // On vide les timers encore en attente (en cas de relance d'un nouveau décompte avant la fin du précédent)
00125         moteur.purge();
00126 
00127         tempsRestant = 0;
00128 
00129         // On exécute la fonction s'il y en a une      
00130         if (fonction != NULL)
00131         {
00132                 //moteur.addTimer(Timer(0.5f,fonction,parametre));
00133                 fonction(parametre);
00134                 // Au cas où finDecompte serait rappelée, on réinitialise l'adresse de la fonction
00135                 fonction = NULL;
